菱形图片

<style>
  body { margin: 0; display: flex; justify-content: center; }
  .diamond {
    width: 120px; height: 120px; margin: 40px 30px;
    display: flex; justify-content: center; align-items: center;
  }
  .diamond-img { width: 120px; height: 120px; margin: 40px 30px; }
  /* 变形方法 */
  .diamond:nth-of-type(1) { transform: rotate(45deg); overflow: hidden; }
  .diamond:nth-of-type(1) > img {
    max-width: 142%; /* 100% * 1.414 向上取整 */
    transform: rotate(-45deg);
  }
  .diamond:nth-of-type(2) { transform: rotate(45deg); overflow: hidden; }
  .diamond:nth-of-type(2) > img {
    max-width: 100%;
    transform: rotate(-45deg) scale(1.42); /* 使用 scale 放大图片 */
  }
  /* 剪切路径方法 (可以很好地适应非正方形的图片) */
  .diamond-img {
    width: 120px;
    height: 120px;
    clip-path: polygon(50% 0, 100% 50%, 50% 100%, 0 50%);
    transform: scale(1.42);
  }
</style>
<body>
  <div class="diamond"><img src="../assets/mao.png"></div>
  <div class="diamond"><img src="../assets/mao.png"></div>
  <img class="diamond-img" src="../assets/mao.png">
</body>
